Math.abs() – funkcija grąžina absoliučią skaičiaus reikšmę.

function difference(a, b) { return Math.abs(a - b); }
console.log(difference(3, 5)); // expected output: 2
console.log(difference(5, 3)); // expected output: 2

Math.ceil() visada suapvalina ir grąžina mažesnį sveikąjį skaičių, didesnį už nurodytą skaičių arba jam lygų.

console.log(Math.ceil(.95));
// expected output: 1
console.log(Math.ceil(4));
// expected output: 4

// 00000000000000000000000000000001

// expected output: 31

Math.floor() - visada apvalina žemyn ir grąžina didžiausią sveikąjį skaičių, mažesnį už nurodytą skaičių arba jam lygų.

console.log(Math.floor(5.95));
// expected output: 5

Math.fround() - grąžina artimiausią 32 bitų vieno tikslumo slankiąją skaičiaus reprezentaciją.

Math.fround(1.5); // 1.5
Math.fround(1.5) === 1.5; // true

Math.hypot() - grąžina savo argumentų kvadratų sumos kvadratinę šaknį. Tai yra,

// expected output: 5

Math.imul() - grąžina dviejų parametrų C tipo 32 bitų daugybos rezultatą.

Math.imul(2, 4); // 8
Math.imul(-1, 8); // -8

Math.log() - grąžina natūralų skaičiaus logaritmą (e bazę).

function getBaseLog(x, y) {
return Math.log(y) / Math.log(x);
} // 2 x 2 x 2 = 8
console.log(getBaseLog(2, 8));
// expected output: 3
// 5 x 5 x 5 x 5 = 625
console.log(getBaseLog(5, 625));
// expected output: 4
